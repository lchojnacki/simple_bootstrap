Sortowanie bąbelkowe

def bubbleSort(alist):
    for passnum in range(len(alist)-1,0,-1):
        for i in range(passnum):
            if alist[i]>alist[i+1]:
                temp = alist[i]
                alist[i] = alist[i+1]
                alist[i+1] = temp
                    
alist = [54,26,93,17,77,31,44,55,20]
bubbleSort(alist)
print(alist)

Sprawdzanie, czy liczba jest pierwsza

while True:
    liczba = int(input("Podaj liczbę: "))
    if liczba > 2:
        i = 2
        while i * i <= liczba:
            if liczba%i == 0:
                print("Nie jest pierwsza")
                break
            i += 1
        else:
            print("Jest pierwsza")

Usuwanie samogłosek z tekstu

samogloski = "aąeęioóuy"
tekst = input("Podaj tekst: ")

nowy_tekst = ''

for litera in tekst:
    if litera.lower() in samogloski:
        continue
    else:
        nowy_tekst += litera

print(nowy_tekst)
